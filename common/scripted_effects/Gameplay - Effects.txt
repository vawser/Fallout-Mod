#All are Country Scope unless otherwise stated.

#Stability
#Country Scope
add_stability_or_adm_power = {
	if = {
		limit = {
			stability = 3
		}
		add_adm_power = 50
	}
	if = {
		limit = {
			NOT = { stability = 3 }
		}
		add_stability = 1
	}
}

spawn_pretender_rebels = {
    spawn_rebels = {
        type = pretender_rebels
        size = $SIZE$
        win = yes
        unrest = $UNREST$
        friend = $TAG$
    }
}
