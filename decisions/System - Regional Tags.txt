	hide_the_formable_states = {
		major = yes
		
		potential = {
			ai = no
			OR = {
				NOT = { has_country_flag = hide_formation_decisions }

			always = yes

			set_country_flag = hide_formation_decisions

			factor = 1

				factor = 0

		}
	}

		}

			ai = no
		}
		
		effect = {
			clr_country_flag = hide_formation_decisions
			set_country_flag = show_formation_decisions
		}

			factor = 0
		}
