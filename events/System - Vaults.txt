# Petition for vault election 
# Some citizens of the vault are requesting permission to engage in an election for the position of overseer. Although many wish to engage in such a process, is it truely wise to undermine the authority of the overseer position? 
country_event = {
	id = vault_government.1
	title = "vault_government.1.title"
	desc = "vault_government.1.desc"
    picture = VAULT_OFFICE_eventPicture

    is_triggered_only = yes
    
	trigger = {
		has_reform = vault_reform
		NOT = { has_ruler_modifier = overseer_for_life }
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
    immediate = {
        hidden_effect = {
            if = { limit = { has_global_flag = enable_event_logging } log = "[Root.GetName] - vault_government.1" }
        }
    }
    
	option = {
		name = vault_governmentS.1.option.a      # The Overseer rules for life!
        
		custom_tooltip = remains_ruler_overseer 
		add_ruler_modifier = {
            name = overseer_for_life
            duration = -1
        }
        
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				NOT = { stability = 1 }
			}
		}
        change_adm = 2
        change_dip = 2
        change_mil = 2
	}	
	
	option = {
		name = vault_governmentS.1.option.b      # A candidate from vault security would secure a strong future

		kill_ruler = yes
		define_ruler = {
			adm = 2
            dip = 2
            mil = 4
            fixed = yes
			age = 30
		}
        
		ai_chance = {
			factor = 15
		}		
	}	
	
	option = {
		name = vault_governmentS.1.option.c      # A candidate from the general population will suffice

		kill_ruler = yes
		define_ruler = {
			adm = 4
            dip = 2
            mil = 2
            fixed = yes
			age = 30
		}
        
		ai_chance = {
			factor = 15
		}		
	}	
	
	option = {
		name = vault_governmentS.1.option.D      # Perhaps this should just be resolved with a bakeoff 

		kill_ruler = yes
		define_ruler = {
			adm = 2
            dip = 4
            mil = 2
            fixed = yes
			age = 30
		}
        
		ai_chance = {
			factor = 15
		}		
	}	
}
